
## Zod Schemas

```typescript
// src/registry/schema.ts
import { z } from 'zod';

export const ClusterEntrySchema = z.object({
  id: z.string().min(1),
  name: z.string().min(1),
  path: z.string().min(1),
  cloudUrl: z.string().url(),
  lastSeen: z.string().datetime(),
});

export const ClusterRegistrySchema = z.object({
  version: z.literal(1),
  clusters: z.array(ClusterEntrySchema),
});

export type ClusterEntry = z.infer<typeof ClusterEntrySchema>;
export type ClusterRegistry = z.infer<typeof ClusterRegistrySchema>;
```

## Validation Rules

| Field | Rule | Error behavior |
|-------|------|----------------|
| `version` | Must be literal `1` | Zod parse failure → return empty registry |
| `id` | Non-empty string | Zod parse failure on load |
| `name` | Non-empty string | Zod parse failure on load |
| `path` | Non-empty string (absolute path) | Resolved via `node:path.resolve()` at query time |
| `cloudUrl` | Valid URL | Zod `.url()` validation |
| `lastSeen` | ISO 8601 datetime | Zod `.datetime()` validation |
| `clusters` | Array (may be empty) | Empty array is valid |
