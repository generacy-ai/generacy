### Humancy Checkpoint Integration

```typescript
// When workflow reaches humancy.request_review step
async function executeHumancyReview(step: WorkflowStep, context: WorkflowContext) {
  const reviewRequest = await humancy.requestReview({
    artifact: interpolate(step.with.artifact, context),
    context: interpolate(step.with.context, context),
    urgency: 'blocking_soon',
  });
  
  // Workflow pauses here
  context.pendingReview = reviewRequest.id;
  await saveWorkflowState(context);
  
  // Wait for human response
  const response = await humancy.waitForReview(reviewRequest.id);
  
  return {
    approved: response.approved,
    comments: response.comments,
  };
}
```
