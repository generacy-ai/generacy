### Workflow Definition

```typescript
interface WorkflowDefinition {
  name: string;
  version: string;
  steps: WorkflowStep[];
  onError?: ErrorHandler;
  timeout?: number;
}

interface WorkflowStep {
  id: string;
  type: 'agent' | 'human' | 'integration' | 'condition';
  config: StepConfig;
  next?: string | ConditionalNext[];
  timeout?: number;
  retries?: number;
}
```
